BAŞLA
    GİRDİ: Şifrelenecek Mesaj (String), Başlangıç Tohumu (Seed - Integer)
    ÇIKTI: Şifrelenmiş Mesaj (Hex String)

    MesajBitleri = StringiBitDizisineÇevir(Mesaj)
    GerekliBitSayisi = MesajBitleri.Uzunluk
    RastgeleAnahtarBitleri = BOŞ LİSTE

    DÖNGÜ (RastgeleAnahtarBitleri.Uzunluk < GerekliBitSayisi):
        Seed = CollatzSonrakiAdim(Seed)
        Val1 = Seed
        Seed = CollatzSonrakiAdim(Seed)
        Val2 = Seed
        
        Bit1 = Val1 MOD 2
        Bit2 = Val2 MOD 2

        EĞER (Bit1 == 0 VE Bit2 == 1) İSE EKLE 1 RastgeleAnahtarBitleri'ne
        EĞER (Bit1 == 1 VE Bit2 == 0) İSE EKLE 0 RastgeleAnahtarBitleri'ne
    DÖNGÜ SONU

    XorlanmisBitler = BOŞ LİSTE
    DÖNGÜ (i = 0'dan GerekliBitSayisi'na kadar):
        YeniBit = MesajBitleri[i] XOR RastgeleAnahtarBitleri[i]
        EKLE YeniBit XorlanmisBitler'e
    DÖNGÜ SONU

    OrtaNokta = XorlanmisBitler.Uzunluk / 2
    SolYari = XorlanmisBitler[0...OrtaNokta-1]
    SagYari = XorlanmisBitler[OrtaNokta...Son]

    SolYari_Kaydirilmis = SolaDaireselKaydir(SolYari, 1 Adım)
    SagYari_Kaydirilmis = SagaDaireselKaydir(SagYari, 1 Adım)

    SifreliBitDizisi = Birlestir(SolYari_Kaydirilmis, SagYari_Kaydirilmis)

    SifreliHex = BitDizisiniHexStringeÇevir(SifreliBitDizisi)
    YAZDIR SifreliHex
BİTİR
